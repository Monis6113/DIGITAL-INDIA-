    function SN_onBlur(val, id) {

        switch (id) {
            case 0:
                classname = 'txtclass0';
                chdname = 'txtname0'

                break;

            case 1:
                classname = 'txtclass1';
                chdname = 'txtname1'

                break;
            case 2:
                classname = 'txtclass2';
                chdname = 'txtname2'

                break;
        }

        if (val != "") {
            $.ajax({
                type: "POST",
                url: "ajaxreq.aspx?AdmID=" + val,
                success: function (data) {
                    if (data.indexOf('||') > 0) {
                        var arrdata = data.split('||')

                        $("#" + classname).val(arrdata[0]);
                        $("#" + chdname).val(arrdata[1]);

                    }
                    else {
                        $("#" + classname).val('');
                        $("#" + chdname).val('');

                    }
                },
                error: function (data) {
                    //alert("error" + data);
                }
            });
        }
        else {
            $("#" + classname).val('');
            $("#" + chdname).val('');

        }
    }
